## Expo Setup

TailSignal works with any app built on Expo or React Native that uses **Expo Push Tokens**.

### Install Dependencies

```bash
npx expo install expo-notifications expo-device expo-constants
```

### Register for Push Notifications

```typescript
import * as Notifications from 'expo-notifications';
import * as Device from 'expo-device';
import Constants from 'expo-constants';
import { Platform } from 'react-native';

async function registerForPushNotifications(wordpressSiteUrl: string) {
    if (!Device.isDevice) {
        console.warn('Push notifications require a physical device');
        return;
    }

    const { status: existingStatus } = await Notifications.getPermissionsAsync();
    let finalStatus = existingStatus;

    if (existingStatus !== 'granted') {
        const { status } = await Notifications.requestPermissionsAsync();
        finalStatus = status;
    }

    if (finalStatus !== 'granted') {
        console.warn('Push notification permission not granted');
        return;
    }

    const projectId = Constants.expoConfig?.extra?.eas?.projectId;
    const token = (await Notifications.getExpoPushTokenAsync({ projectId })).data;

    // Register with TailSignal
    await fetch(`${wordpressSiteUrl}/wp-json/tailsignal/v1/register`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
            expo_token: token,
            device_type: Platform.OS,
            device_model: Device.modelName,
            os_version: `${Platform.OS} ${Device.osVersion}`,
            app_version: Constants.expoConfig?.version || '1.0.0',
            user_label: `${Device.deviceName}`,
        }),
    });

    return token;
}
```

### Unregister Device

```typescript
async function unregisterDevice(wordpressSiteUrl: string, token: string) {
    await fetch(`${wordpressSiteUrl}/wp-json/tailsignal/v1/register`, {
        method: 'DELETE',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ expo_token: token }),
    });
}
```

## Handling Notifications

### Notification Data

When TailSignal sends a notification for a published post, the data payload includes:

```json
{
    "post_id": 123,
    "post_type": "post",
    "url": "https://example.com/my-post/"
}
```

For portfolio items, `post_type` will be `"portfolio"`. Use this to route users to the correct screen in your app.

### Notification Listener

```typescript
import * as Notifications from 'expo-notifications';

// Handle notification tapped (app opened from notification)
Notifications.addNotificationResponseReceivedListener((response) => {
    const data = response.notification.request.content.data;

    if (data.post_id) {
        if (data.post_type === 'portfolio') {
            // Navigate to portfolio screen
            navigation.navigate('Portfolio', { id: data.post_id });
        } else {
            // Navigate to blog post screen
            navigation.navigate('BlogPost', { id: data.post_id });
        }
    } else if (data.url) {
        // Open web link
        Linking.openURL(data.url);
    }
});
```

## Android Notification Channels

For Android, set up a default notification channel:

```typescript
if (Platform.OS === 'android') {
    Notifications.setNotificationChannelAsync('default', {
        name: 'Default',
        importance: Notifications.AndroidImportance.MAX,
        vibrationPattern: [0, 250, 250, 250],
        lightColor: '#0FACED',
    });
}
```
